Assert.js

The assertion library for Javascript.

View on GitHub

In order to make sure that state is always consistent and predictable you can use strongly typed language or write unit tests. AssertJS stands between those choices. It can be used inside of unit tests as well as inside of application model.

Installation

NPM

npm install assert-js --save

Browser

<script src="http://assert-js.com/release/0.6.0/assert.min.js" type="text/javascript"></script>

Usage

NPM

var Assert = require('assert-js');

Browser

var Assert = window.Assert;

Available assertions

Assert.instanceOf(new String("test"), String);
Assert.containsOnly([new String("test"), new String("test")],String);
Assert.integer(1);
Assert.number(0.5);
Assert.oddNumber(3);
Assert.evenNumber(4);
Assert.greaterThan(1, 10)
Assert.greaterThanOrEqual(1, 1)
Assert.lessThan(10, 5)
Assert.lessThanOrEqual(1, 1)
Assert.string("string");
Assert.boolean(true);
Assert.object({id: 1});
Assert.hasFunction("testFunction", {testFunction: () => { alert('test')' });
Assert.isFunction(() => { alert('test'); });
Assert.array([1, 2, 3, 4, 5]);
Assert.count(0, []);
Assert.notEmpty(0, [1, 2, 3]);
Assert.jsonString('{"key": "value"}');
Assert.email('norbert@orzechowicz.pl');
Assert.url('https://github.com/Tiliqua/assert-js');
Assert.uuid('3e9009a0-4b2f-414e-bf02-ec0df56fc864');

Something is missing? Don't worry, AssertJS is Open Source project at MIT license. Feel free to contribute!

Custom exception message

Assert.string(1234, 'custom error message');
Assert.string(1234, 'Expected ${expected} but got ${received}');
// it throws "Expected string but got int[1234]"
